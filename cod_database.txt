    IDCOSTUMER INT NOT NULL AUTO_INCREMENT,
    NAME VARCHAR(50) NOT NULL,

    STREET VARCHAR(50) NOT NULL,
    NUMBER VARCHAR(5) NOT NULL,
    DISTRICT VARCHAR(50) NOT NULL,
